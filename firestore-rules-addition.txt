// News app collections — add inside match /databases/{database}/documents { }

// CLA historical data — all authenticated users can read; admin writes via script
match /cla_historical/{period} {
  allow read: if request.auth != null;
  allow write: if request.auth != null;
}

// CLA meta analysis reports — all authenticated users can read; admin writes
match /cla_meta/{region} {
  allow read: if request.auth != null;
  allow write: if request.auth != null;
}

// Comments on news articles — all authenticated users can read; owners can CUD
match /news_comments/{commentId} {
  allow read: if request.auth != null;
  allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
  allow update, delete: if request.auth != null && resource.data.userId == request.auth.uid;
}

// Bookmarks — private per user
match /news_bookmarks/{bookmarkId} {
  allow read: if request.auth != null && resource.data.userId == request.auth.uid;
  allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
  allow delete: if request.auth != null && resource.data.userId == request.auth.uid;
}

// Foresight queries — private per user
match /foresight_queries/{queryId} {
  allow read: if request.auth != null && resource.data.userId == request.auth.uid;
  allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
  allow delete: if request.auth != null && resource.data.userId == request.auth.uid;
}

// User-generated reports — private per user
match /user_reports/{reportId} {
  allow read: if request.auth != null && resource.data.userId == request.auth.uid;
  allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
  allow delete: if request.auth != null && resource.data.userId == request.auth.uid;
}

// Insight report admin config — authenticated users can read/write
match /insight_report_config/{docId} {
  allow read, write: if request.auth != null;
}

// Insight report status/edits — authenticated users can read/write
match /insight_report_status/{docId} {
  allow read, write: if request.auth != null;
}
match /insight_report_edits/{docId} {
  allow read, write: if request.auth != null;
}
